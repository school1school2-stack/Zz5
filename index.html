// 1. بصمة الجهاز المطورة (تجمع معلومات أكثر لضمان عدم التكرار) function getFingerprint() { const info = [ navigator.userAgent, screen.width + "x" + screen.height, navigator.language, new Date().getTimezoneOffset(), navigator.hardwareConcurrency || 'unknown' ].join('|'); return btoa(info); } const deviceID = getFingerprint(); async function secureStart() { const nameInput = document.getElementById('studentName').value.trim(); const classInput = document.getElementById('studentClass').value; // التحقق من الاسم if(nameInput.split(' ').length < 3) return alert("الاسم الثلاثي مطلوب."); if(!classInput) return alert("اختر الفصل."); document.getElementById('start-btn').disabled = true; document.getElementById('checking').classList.remove('hidden'); // 2. فحص الحماية المحلية (LocalStorage) لمنع التكرار في نفس المتصفح حتى بدون إنترنت const localCheck = localStorage.getItem(`done_${currentModelId}`); // 3. فحص قاعدة البيانات (الفحص الأقوى لمنع التحايل عبر أجهزة أخرى أو متصفحات مختلفة بنفس البصمة) const q = await db.collection("student_results") .where("deviceID", "==", deviceID) .where("model_id", "==", currentModelId) .get(); if(!q.empty || localCheck) { document.getElementById('start-page').classList.add('hidden'); document.getElementById('lock-screen').classList.remove('hidden'); document.getElementById('checking').classList.add('hidden'); return; } // إذا اجتاز الفحص، نبدأ الاختبار sName = nameInput; sClass = classInput; document.getElementById('start-page').classList.add('hidden'); document.getElementById('quiz-page').classList.remove('hidden'); loadQuestion(); } async function finish() { document.getElementById('quiz-page').classList.add('hidden'); document.getElementById('result-page').classList.remove('hidden'); document.getElementById('score-text').innerText = `${userScore} / ${quizData.length}`; // حفظ علامة في المتصفح الحالي localStorage.setItem(`done_${currentModelId}`, "true"); // حفظ النتيجة في Firebase مع بصمة الجهاز للتأكد من عدم دخوله مرة أخرى await db.collection("student_results").add({ name: sName, student_class: sClass, score: userScore, total: quizData.length, model_id: currentModelId, deviceID: deviceID, time: firebase.firestore.FieldValue.serverTimestamp() }); document.getElementById('cloud-msg').innerText = "تم حفظ النتيجة بنجاح. لا يمكنك دخول الاختبار مرة أخرى."; }
